00086         override protected void TransformRow (float[,] src, float[,] dst, int y, int length)
00087         {
00088             if (length >= AllowedMinSize) {
00089                 int half = length >> 1;
00090                 //if the length is even then subtract 1 from "half"
00091                 //as there is the same number of low and high-pass values
00092                 //(Note that "numLFValues" is equal to "half+1") 
00093                 //For a odd length there is one additional quency value (so do not subtract 1)
00094                 //"half" is one less than "numLFValues" as we cannot directly
00095                 //calculate the last value in the for-loop (array bounds)
00096                 if ((length & 1) == 0)
00097                     half--;
00098                 int offSrc = 0;
00099                 // starting offset for high-pass values (= number of low-pass values)
00100                 int offdst = half + 1; 
00101                 int numLFValues = offdst;
00102     
00103                 float lastHF = 0.0f;
00104                 for (int i = 0; i < half; i++) {
00105                     //calculate the high-pass value by
00106                     //subtracting the mean of the immediate neighbors for every second value
00107                     float hf = src [offSrc + 1, y] - (src [offSrc, y] + src [offSrc + 2, y]) * Mean;
00108                     //smoothing the low-pass value, scale by factor 2 
00109                     //(instead of scaling low frequencies by factor sqrt(2) and
00110                     //shrinking high frequencies by factor sqrt(2)
00111                     //and reposition to have all low frequencies on the left side
00112                     dst [i, y] = (src [offSrc, y] + (lastHF + hf) * Smooth) * Scale;
00113                     dst [offdst++, y] = hf;
00114                     lastHF = hf;
00115                     offSrc += 2; 
00116                 } 
00117                 if ((length & 1) == 0) {
00118                     //the secound last value in the array is our last low-pass value
00119                     dst [numLFValues - 1, y] = Scale * src [length - 2, y]; 
00120                     //the last value is a high-pass value
00121                     //however here we just subtract the previos value (so not really a
00122                     //biorthogonal 5/3 transformation)
00123                     //This is done because the 5/3 wavelet cannot be calculated at the boundary
00124                     dst [length - 1, y] = src [length - 1, y] - src [length - 2, y];
00125                 } else {
00126                     dst [numLFValues - 1, y] = Scale * src [length - 1, y]; 
00127                 }
00128             } else {
00129                 //We cannot perform the biorthogonal 5/3 wavelet transformation
00130                 //for lengths smaller than 3. We could do a simpler transformation...
00131                 //Here however, we just copy the values from the source to the destination array  
00132                 for (int i = 0; i < length; i++)
00133                     dst [i, y] = src [i, y];
00134             }

00205                 float lastHF = src [numLFValues, y];
00206                 //Calculate the first two values outside the for loop (array bounds)

00209                 for (int i = 1; i < half; i++) {
00210                     float hf = src [numLFValues + i, y];
00211                     float lf = InvScale * src [i, y];
00212                     //reconstruct the original value by removing the "smoothing" 
00213                     float lfReconst = lf - (hf + lastHF) * Smooth;
00214                     dst [2 * i, y] = lfReconst;
00215                     //add reconstructed low-pass value (left side) and high-pass value
00216                     dst [2 * i + 1, y] = lfReconst * Mean + hf;

00219                     //reconstructed values is not known earlier
00220                     dst [2 * i - 1, y] += lfReconst * Mean;
