00065         override protected void TransformRow (float[,] src, float[,] dst, int y, int length)
00066         {
00067             if (length >= allowedMinSize) {
00068                 int half = length >> 1;
00069                 int offSrc = 0;
00070                 // number of low-pass values
00071                 int numLFValues = half + (length & 1);
00072 
00073                 for (int i = 0; i < half; i++) {
00074                     float a = src [offSrc    , y];
00075                     float b = src [offSrc + 1, y];
00076                     //calculate the mean of a and b and scale with factor 2
00077                     //So no multiplication needed at all
00078                     dst [i              , y] = (a + b);
00079                     dst [i + numLFValues, y] = (b - a) * MEAN;
00080                     offSrc += 2;
00081                 }                           
00082                 if ((length & 1) != 0) {
00083                     dst [numLFValues - 1, y] = src [length - 1, y] * SCALE;
00084                 }       
00085             } else {
00086                 for (int i = 0; i < length; i++)
00087                     dst [i, y] = src [i, y];
00088             }
00089         }
