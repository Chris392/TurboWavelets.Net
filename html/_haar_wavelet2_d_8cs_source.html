00080         override protected void TransformRow (float[,] src, float[,] dst, int y, int length)
00081         {
00082             if (length >= AllowedMinSize) {
00083                 int half = length >> 1;
00084                 int offSrc = 0;
00085                 // number of low-pass values
00086                 int numLFValues = half + (length & 1);
00087 
00088                 for (int i = 0; i < half; i++) {
00089                     float a = src [offSrc    , y];
00090                     float b = src [offSrc + 1, y];
00091                     //calculate the mean of a and b and scale with factor 2
00092                     //So no multiplication needed at all
00093                     dst [i              , y] = (a + b);
00094                     dst [i + numLFValues, y] = (b - a) * Mean;
00095                     offSrc += 2;
00096                 }                           
00097                 if ((length & 1) != 0) {
00098                     dst [numLFValues - 1, y] = src [length - 1, y] * Scale;
00099                 }       
00100             } else {
00101                 for (int i = 0; i < length; i++)
00102                     dst [i, y] = src [i, y];
00103             }
