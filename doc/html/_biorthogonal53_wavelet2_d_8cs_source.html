62  override protected void TransformRow (float[,] src, float[,] dst, int y, int length)
63  {
64  if (length >= MINSIZE) {
65  int half = length >> 1;
66  //if the length is even then subtract 1 from "half"
67  //as there is the same number of low and high-pass values
68  //(Note that "numLFValues" is equal to "half+1")
69  //For a odd length there is one additional quency value (so do not subtract 1)
70  //"half" is one less than "numLFValues" as we cannot directly
71  //calculate the last value in the for-loop (array bounds)
72  if ((length & 1) == 0)
73  half--;
74  int offSrc = 0;
75  // starting offset for high-pass values (= number of low-pass values)
76  int offdst = half + 1;
77  int numLFValues = offdst;
78 
79  float lastHF = 0.0f;
80  for (int i = 0; i < half; i++) {
81  //calculate the high-pass value by
82  //subtracting the mean of the immediate neighbors for every second value
83  float hf = src [offSrc + 1, y] - (src [offSrc, y] + src [offSrc + 2, y]) * MEAN;
84  //smoothing the low-pass value, scale by factor 2
85  //(instead of scaling low frequencies by factor sqrt(2) and
86  //shrinking high frequencies by factor sqrt(2)
87  //and reposition to have all low frequencies on the left side
88  dst [i, y] = SCALE * (src [offSrc, y] + (lastHF + hf) * SMOOTH);
89  dst [offdst++, y] = hf;
90  lastHF = hf;
91  offSrc += 2;
92  }
93  if ((length & 1) == 0) {
94  //the secound last value in the array is our last low-pass value
95  dst [numLFValues - 1, y] = SCALE * src [length - 2, y];
96  //the last value is a high-pass value
97  //however here we just subtract the previos value (so not really a
98  //biorthogonal 5/3 transformation)
99  //This is done because the 5/3 wavelet cannot be calculated at the boundary
100  dst [length - 1, y] = src [length - 1, y] - src [length - 2, y];
101  } else {
102  dst [numLFValues - 1, y] = SCALE * src [length - 1, y];
103  }
104  } else {
105  //We cannot perform the biorthogonal 5/3 wavelet transformation
106  //for lengths smaller than 3. We could do a simpler transformation...
107  //Here however, we just copy the values from the source to the destination array
108  for (int i = 0; i < length; i++)
109  dst [i, y] = src [i, y];
110  }

165  override protected void InvTransformRow (float[,] src, float[,] dst, int y, int length)
166  {
167  if (length >= MINSIZE) {
168  int half = length >> 1;
169  //if the length is even then subtract 1 from "half"
170  //as there is the same number of low and high-pass values
171  //(Note that "numLFValues" is equal to "half+1")
172  //For a odd length there is one additional low-pass value (so do not subtract 1)
173  //"half" is one less than "numLFValues" as we cannot directly
174  //calculate the last value in the for-loop (array bounds)
175  if ((length & 1) == 0)
176  half--;
177  // number of low-pass values
178  int numLFValues = half + 1;
179 
180  float lastLF = SCALE_INV * src [0, y] - src [numLFValues, y] * SMOOTH;
181  float lastHF = src [numLFValues, y];
182  //Calculate the first two values outside the for loop (array bounds)
183  dst [0, y] = lastLF;
184  dst [1, y] = lastHF + lastLF * SCALE_INV;
185  for (int i = 1; i < half; i++) {
186  float hf = src [numLFValues + i, y];
187  float lf = SCALE_INV * src [i, y];
188  //reconstruct the original value by removing the "smoothing"
189  float lfReconst = lf - (hf + lastHF) * SMOOTH;
190  dst [2 * i, y] = lfReconst;
191  //add reconstructed low-pass value (left side) and high-pass value
192  dst [2 * i + 1, y] = lfReconst * MEAN + hf;
193  //add other low-pass value (right side)
194  //This must be done one iteration later, as the
195  //reconstructed values is not known earlier
196  dst [2 * i - 1, y] += lfReconst * MEAN;
197  lastHF = hf;
198  lastLF = lfReconst;
199  }
200 
201  if ((length & 1) == 0) {
202  //restore the last 3 values outside the for loop
203  //adding the missing low-pass value (right side)
204  dst [length - 3, y] += src [numLFValues - 1, y] * MEAN * SCALE_INV;
205  //copy the last low-pass value
206  dst [length - 2, y] = src [numLFValues - 1, y] * SCALE_INV;
207  //restore the last value by adding last low-pass value
208  dst [length - 1, y] = src [length - 1, y] + src [numLFValues - 1, y] * SCALE_INV;
209  } else {
210  //restore the last 2 values outside the for loop
211  //adding the missing low-pass value (right side)
212  dst [length - 2, y] += src [numLFValues - 1, y] * MEAN * SCALE_INV;
213  //copy the last low-pass value
214  dst [length - 1, y] = src [numLFValues - 1, y] * SCALE_INV;
215  }
216  } else {
217  //We cannot perform the biorthogonal 5/3 wavelet transformation
218  //for lengths smaller than 3. We could do a simpler transformation...
219  //Here however, we just copy the values from the source to the destination array
